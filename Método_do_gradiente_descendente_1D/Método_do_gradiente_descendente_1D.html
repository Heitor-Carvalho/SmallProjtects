# ImplementaÃ§Ã£o mÃ©todo gradiente descendente
def gradient_descent(fx, gradx, x0, delta_lim, alpha, max_it = 300):
    x_steps = [x0]
    fx_steps = [f(x0)]
    
    delta = f(x0)
    x = x0
    it = 0
    while np.abs(delta) > delta_lim and it < max_it:
        x_prev = x
        x -= alpha*gradx(x)
        delta = x_prev - x
        it += 1
        x_steps.append(x)
        fx_steps.append(f(x))
    
    return x_steps, fx_steps
